1.Right a Program to reverse a string using C# Code

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter a string: ");
        string input = Console.ReadLine();

        string reversed = ReverseString(input);

        Console.WriteLine($"Reversed string: {reversed}");
    }

    static string ReverseString(string str)
    {
        char[] charArray = str.ToCharArray();
        Array.Reverse(charArray);
        return new string(charArray);
    }
}
2.Write a program to Count the char from a string
using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        Console.Write("Enter a string: ");
        string input = Console.ReadLine();

        Dictionary<char, int> charCount = new Dictionary<char, int>();

        foreach (char c in input)
        {
            if (charCount.ContainsKey(c))
                charCount[c]++;
            else
                charCount[c] = 1;
        }

        Console.WriteLine("\nCharacter Frequencies:");
        foreach (var kvp in charCount)
        {
            Console.WriteLine($"'{kvp.Key}': {kvp.Value}");
        }
    }
}


3. Write a Program to find the duplicate Element from an arrY

using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        int[] array = { 4, 2, 7, 4, 9, 2, 1, 7, 5 };

        HashSet<int> seen = new HashSet<int>();
        HashSet<int> duplicates = new HashSet<int>();

        foreach (int num in array)
        {
            if (!seen.Add(num))
            {
                duplicates.Add(num);
            }
        }

        Console.WriteLine("Duplicate elements:");
        foreach (int dup in duplicates)
        {
            Console.WriteLine(dup);
        }
    }
}

4. Grouping with LINQ

using System;
using System.Collections.Generic;
using System.Linq;

class Person
{
    public string Name { get; set; }
    public string City { get; set; }
}

class Program
{
    static void Main()
    {
        List<Person> people = new List<Person>
        {
            new Person { Name = "Alice", City = "Mumbai" },
            new Person { Name = "Bob", City = "Delhi" },
            new Person { Name = "Charlie", City = "Mumbai" },
            new Person { Name = "David", City = "Bangalore" },
            new Person { Name = "Eve", City = "Delhi" }
        };

        var groupedByCity = people.GroupBy(p => p.City);

        foreach (var group in groupedByCity)
        {
            Console.WriteLine($"City: {group.Key}");
            foreach (var person in group)
            {
                Console.WriteLine($" - {person.Name}");
            }
        }
    }
}


Example: Group by Category and Sum Amounts
using System;
using System.Collections.Generic;
using System.Linq;

class Transaction
{
    public string Category { get; set; }
    public double Amount { get; set; }
}

class Program
{
    static void Main()
    {
        List<Transaction> transactions = new List<Transaction>
        {
            new Transaction { Category = "Food", Amount = 150 },
            new Transaction { Category = "Travel", Amount = 300 },
            new Transaction { Category = "Food", Amount = 100 },
            new Transaction { Category = "Bills", Amount = 200 },
            new Transaction { Category = "Travel", Amount = 50 }
        };

        var groupedTotals = transactions
            .GroupBy(t => t.Category)
            .Select(g => new
            {
                Category = g.Key,
                TotalAmount = g.Sum(t => t.Amount),
                Count = g.Count()
            });

        foreach (var group in groupedTotals)
        {
            Console.WriteLine($"Category: {group.Category}, Total: {group.TotalAmount}, Count: {group.Count}");
        }
    }
}

C# Program: Group Employees by Department, Count Members, and Average Salary


using System;
using System.Collections.Generic;
using System.Linq;

class Employee
{
    public string Name { get; set; }
    public string Department { get; set; }
    public double Salary { get; set; }
}

class Program
{
    static void Main()
    {
        List<Employee> employees = new List<Employee>
        {
            new Employee { Name = "Alice", Department = "HR", Salary = 50000 },
            new Employee { Name = "Bob", Department = "IT", Salary = 70000 },
            new Employee { Name = "Charlie", Department = "HR", Salary = 55000 },
            new Employee { Name = "David", Department = "Finance", Salary = 60000 },
            new Employee { Name = "Eve", Department = "IT", Salary = 72000 },
            new Employee { Name = "Frank", Department = "IT", Salary = 68000 }
        };

        var grouped = employees
            .GroupBy(e => e.Department)
            .Select(g => new
            {
                Department = g.Key,
                Count = g.Count(),
                AverageSalary = g.Average(e => e.Salary)
            })
            .OrderByDescending(g => g.Count);

        Console.WriteLine("Department Summary:\n");
        foreach (var group in grouped)
        {
            Console.WriteLine($"Department: {group.Department}, Count: {group.Count}, Avg Salary: {group.AverageSalary:C}");
        }
    }
}

{
    ("Alice", "Delhi"), ("Bob", "Mumbai"), ("Charlie", "Delhi")
};

Dictionary<string, List<string>> grouped = new Dictionary<string, List<string>>();
foreach (var person in people)
{
    if (!grouped.ContainsKey(person.City))
        grouped[person.City] = new List<string>();
    grouped[person.City].Add(person.Name);
}

int[] nums = { 1, 2, 3, 2, 4, 1 };
Dictionary<int, int> count = new Dictionary<int, int>();
foreach (int num in nums)
{
    if (count.ContainsKey(num))
        count[num]++;
    else
        count[num] = 1;
}
var duplicates = count.Where(kvp => kvp.Value > 1).Select(kvp => kvp.Key);
